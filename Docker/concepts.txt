//Docker Commands
docker pull IMAGE_NAME
docker pull IMAGE_NAME:version
docker images
docker run -it IMAGE_NAME    
docker stop CONT_NAME or CONT_ID
docker start CONT_NAME or CONT_ID

docker ps -a   //for all container list
docker ps      //only running container list

docker run -d IMAGE_NAME
docker run --name CONT_NAME -d IMAGE_NAME



IF WANT to use define env variables : -e
example: docker run -d -e MYSQL_ROOT_PASSWORD=secret mysql
